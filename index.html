class JuegoAhorcado:
    def __init__(self):
        self.palabra_secreta = ""
        self.letras_adivinadas = []
        self.letras_incorrectas = []
        self.intentos_restantes = 6
        self.estado_juego = "jugando"
        self.diccionario = DiccionarioPalabras()
        self.interfaz = InterfazConsola()
    
    def iniciar_juego(self):
        """Inicia una nueva partida del juego"""
        self.palabra_secreta = self.diccionario.obtener_palabra_aleatoria()
        self.letras_adivinadas = []
        self.letras_incorrectas = []
        self.intentos_restantes = 6
        self.estado_juego = "jugando"
        self.interfaz.mostrar_mensaje_bienvenida()
    
    def procesar_letra(self, letra):
        """Procesa la letra ingresada por el usuario"""
        if letra in self.letras_adivinadas or letra in self.letras_incorrectas:
            return "Ya has usado esta letra"
        
        if letra in self.palabra_secreta:
            self.letras_adivinadas.append(letra)
            if self._palabra_completa():
                self.estado_juego = "ganado"
        else:
            self.letras_incorrectas.append(letra)
            self.intentos_restantes -= 1
            if self.intentos_restantes == 0:
                self.estado_juego = "perdido"
    
    def _palabra_completa(self):
        """Verifica si la palabra ha sido completamente adivinada"""
        return all(letra in self.letras_adivinadas for letra in self.palabra_secreta)
    
    def obtener_palabra_mostrar(self):
        """Retorna la palabra con las letras adivinadas visibles"""
        return ''.join(letra if letra in self.letras_adivinadas else '_' 
                      for letra in self.palabra_secreta)

import random
import os

class DiccionarioPalabras:
    def __init__(self, archivo_diccionario="palabras.txt"):
        self.archivo_diccionario = archivo_diccionario
        self.palabras = []
        self.categorias = {
            "facil": [],
            "medio": [],
            "dificil": []
        }
        self.cargar_palabras()
    
    def cargar_palabras(self):
        """Carga las palabras desde el archivo de diccionario"""
        try:
            if os.path.exists(self.archivo_diccionario):
                with open(self.archivo_diccionario, 'r', encoding='utf-8') as archivo:
                    self.palabras = [palabra.strip().upper() 
                                   for palabra in archivo.readlines() 
                                   if palabra.strip()]
                self._categorizar_palabras()
            else:
                self._crear_diccionario_por_defecto()
        except Exception as e:
            print(f"Error al cargar el diccionario: {e}")
            self._crear_diccionario_por_defecto()
    
    def _categorizar_palabras(self):
        """Categoriza las palabras por dificultad seg煤n su longitud"""
        for palabra in self.palabras:
            if len(palabra) <= 4:
                self.categorias["facil"].append(palabra)
            elif len(palabra) <= 7:
                self.categorias["medio"].append(palabra)
            else:
                self.categorias["dificil"].append(palabra)
    
    def obtener_palabra_aleatoria(self, categoria="todas"):
        """Obtiene una palabra aleatoria del diccionario"""
        if categoria == "todas":
            return random.choice(self.palabras) if self.palabras else "PYTHON"
        else:
            palabras_categoria = self.categorias.get(categoria, self.palabras)
            return random.choice(palabras_categoria) if palabras_categoria else "PYTHON"
    
    def _crear_diccionario_por_defecto(self):
        """Crea un diccionario b谩sico si no existe el archivo"""
        palabras_por_defecto = [

            "VARIABLE", "FUNCION", "CLASE", "OBJETO", "METODO",
            "HERENCIA", "POLIMORFISMO", "ENCAPSULACION"
